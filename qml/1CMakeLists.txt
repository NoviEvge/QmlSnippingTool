qt_add_qml_module( qmlProject
    URI ${CMAKE_PROJECT_NAME}
    VERSION 1.0
    QML_FILES
        main.qml
		ActionButtons.qml
        DelayBox.qml
		Properties.qml
		SaveFileDialog.qml
		ScreenshotWindow.qml
		Sequences.qml
		SettingsWindow.qml
		Support.qml
    RESOURCES
        images/clipboard.svg
		images/reset.svg
		images/save.svg
		images/scissors.svg
		images/settings.svg
)

find_package(Qt${INSTALLED_QT_VERSION} COMPONENTS Qml REQUIRED)
target_link_libraries(${CMAKE_PROJECT_NAME} Qt${INSTALLED_QT_VERSION}::Qml)
